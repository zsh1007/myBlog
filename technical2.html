mui事件封装成angular的指令

ngModule.directive('ngTap', ['$parse',

    function($parse) {

        return {

            restrict: 'A',

            compile: function($element, attr) {

                var fn = $parse(attr["ngTap"]);

                return function ngEventHandler(scope, elem) {

                    elem[0].addEventListener("tap", function(e) {

                        var callback = function(e) {

                            fn(scope, {

                                $event: e

                            });

                        };

                        scope.$apply(callback);

                    });

                };

            }

        };

    }

]);


mui+angular中，mui的事件在angular中无法触发脏检查，所以通过angualr的指令将mui的tap等事件封装成指令，将定义的变量绑定到watch队列，使得数据可以双向绑定。
